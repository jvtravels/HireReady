                display: "flex", alignItems: "center", justifyContent: "center",

      {showUpgradeModal && (
        <UpgradeModal

          sessionsUsed={freeSessionCount}
          user={user}
          currentTier={user?.subscriptionTier || "free"}
          onPaymentSuccess={(_tier: string, _start: string, _end: string) => {
